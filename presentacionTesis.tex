	\section{Introducción}

		\begin{frame}{¿Qué es el bloqueo de nervios y por qué es importante?}
			\begin{block}{Bloqueo de nervios}
				El bloqueo de nervios es una estrategia utilizada en la anestesiología para el \alert{tratamiento del dolor} en: Operaciones médicas, oncología, etapas post operatorias y por otros casos clínicos \cite{Perez2011} \cite{AcedoGutierrez2005}.
			\end{block}
		
			\begin{figure}[!htb]
				\centering
				\includegraphics[width=0.35\linewidth]{img/bloqueo}
				\caption{Ilustración de bloqueo para el nervio cervical \cite{bloqueo}.}
			\end{figure}
		\end{frame}
	
		\begin{frame}{Técnicas empleadas para efectuar el bloqueo de nervios}
			\begin{block}{Estimulación nerviosa}
				Se utiliza una aguja cuya punta propaga a través de los órganos una \alert{señal eléctrica} con el fin de producir una \alert{respuesta motora} (RM) al estimular un nervio.
			\end{block}
			\begin{figure}
				\centering
				\subfloat[Sin RM. Señal: 1mA y 0.1ms]{
					\label{f:gato}
					\includegraphics[width=0.3\textwidth]{img/estimulacionUno}}
				\subfloat[RM Leve. Señal: 3mA y 0.1ms]{
					\label{f:tigre}
					\includegraphics[width=0.3\textwidth]{img/estimulacionDos}}
				\subfloat[RM mayor. Señal: 3mA y 0.3ms]{
					\label{f:conejo}
					\includegraphics[width=0.3\textwidth]{img/estimulacionTres}}
				\caption{Localización de un nervio a 2cm de la aguja \cite{estimulacion}.}
				\label{f:animales}
			\end{figure}
		\end{frame}
	
		\begin{frame}{Técnicas empleadas para efectuar el bloqueo de nervios}
			\begin{block}{Guiado por ultrasonido}
				El bloqueo de nervios guiado utilizando imágenes de ultrasonido permite identificar elementos como  arterías y venas que deben ser \alert{evitados} y sirven para \alert{guiar} al especialista a encontrar el nervio objetivo \cite{bloqueoUltrasonido}.
			\end{block}
			\begin{center}
				\animategraphics[autoplay,loop,width=0.8\linewidth]{10}{img/gif/frame-}{0}{43}
			\end{center}
			
	\end{frame}
		\begin{frame}{Retos de identificar estructuras nerviosas en imágenes de ultrasonido}
			\begin{block}{Retos}
				\begin{itemize}
					\item La \alert{estructura} y \alert{forma} del nervio presentan no linealidades, además la calidad de las imágenes se ve afectada por \alert{ruido acústico} y las características del transductor.
					\item La implementación de imágenes de ultrasonido ha permitido que expertos en esta técnica logren hasta un \alert{95\% de aciertos} en el bloqueo de nervios.
					\item  Los errores podrían conllevar a \alert{diversas complicaciones} en el paciente como lesiones vasculares, articulares, perdida temporal de caminar normalmente, entre otras\cite{Ivan2006} \cite{GiraldoGutierres2015}.
				\end{itemize}
			\end{block}
		\end{frame}

		\begin{frame}{Segmentación de nervios periféricos utilizando aprendizaje de máquina}
			Con el fin de \alert{asistir en la localización} de estructuras nerviosas en imágenes de ultrasonido, se han desarrollado algoritmos que implementan técnicas de aprendizaje de maquina.
			\begin{block}{Técnicas utilizadas para segmentar nervios periféricos en imágenes de ultrasonido}
				\begin{itemize}
					\item Redes Neuronales Convolucionales.
					\item Enfoques Bayesianos.
					\item Procesos Gausianos.
					\item \alert{Redes Generativas Adversarias}
					\item Filtro + Enfoque Bayesiano
				\end{itemize}
			\end{block}
		\end{frame}
	
		\begin{frame}{Ventajas de implementar Redes Generativas Adversarias}
				Mediante el entrenamiento adversario se consideran características como la \alert{variabilidad anatómica} del nervio y se tiene un \alert{contexto global} de la imagen.
			\begin{figure}[!htb]
				\centering

			\end{figure} 
		\end{frame}

	\section{Objetivos}
	
		\begin{frame}{Objetivos}
			\begin{block}{Objetivo General}
				Desarrollar un sistema automático para la segmentación de nervios periféricos en imágenes de ultrasonido que permita asistir en su identificación utilizando redes generativas adversarias.
			\end{block}
		\end{frame}
	
		\begin{frame}{Objetivos}
			\begin{block}{Objetivos Especificos}
				\begin{enumerate}
					\item Construir un modelo de aprendizaje generativo adversario que permita generar un mapa de segmentación a partir de una imagen de ultrasonido cuantificando su exactitud por medio de funciones de perdida. 
					\item Desarrollar una estrategia de ajuste que utilice enfoques generativos adversarios para aprender la variabilidad de las texturas anatómicas.  
					\item Evaluar la exactitud del sistema automático para identificar los nervios periféricos mediante métodos cuantitativos y compararlos con enfoques comúnmente utilizados en el estado del arte.
				\end{enumerate}
			\end{block}
		\end{frame}

	\section{Conclusiones}
		
		\begin{frame}{Conclusiones}
			\begin{block}{Conclusiones}
				\begin{itemize}
					\item La arquitectura CGAN propuesta permite entrenar modelos para la segmentación de nervios periféricos usando una metodología de entrenamiento adversario condicional.
					\item Las funciones de perdida utilizadas permiten evaluar la calidad de las segmentaciones realizadas por la red generadora durante el proceso de entrenamiento, obteniendo los mejores resultados sobre las métricas de evaluación

			\end{block}
